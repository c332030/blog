[source,sql]
----
SELECT
        COUNT(*)
FROM
    AUTH_USER u
    LEFT JOIN
        AUTH_USER_ROLE_RELATION rr
            ON u.ID = rr.USER_ID
    LEFT JOIN
        AUTH_ROLE r
            ON rr.ROLE_ID = r.ID
    WHERE
        u.CREATOR_USER_ACCOUNT='shunde123'
    GROUP BY
        u.ID
----
